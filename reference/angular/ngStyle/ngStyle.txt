
// Example 1:
// *********** html file ***********
<section class="box">
    <div class="box-view">
        <!--
          1. here we use [ngStyle] directive. with this directive we can 
             change style property of element.
        -->
        <div [ngStyle]="{
                backgroundColor: changeBg ? 'blue' : 'unset',
                color: changeBg ? 'white' : 'blue'
            }"
            class="box-view-bg">box</div>
    </div>
    <div class="button">
        <!--
          1. here we use 'toggle' changes. When user click current button,
             booleat value of 'changeBg' changes to opposite.
        -->
        <button (click)="changeBg = !changeBg">Change Background</button>
    </div>
</section>







// ************ js file *************
import { Component, OnInit } from '@angular/core';

  selector: 'app-background',
  templateUrl: './background.component.html',
  styleUrls: ['./background.component.css']
})
export class BackgroundComponent implements OnInit {
  // we declare new boolean property, with 'false' default value.
  changeBg: boolean = false;
  
  constructor() { }

  ngOnInit(): void {
  }

}
